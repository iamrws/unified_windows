0033:     def _make_policy(self, **policy_overrides):
0034:         kwargs = dict(policy_overrides)
0035:         params = signature(self.SecurityPolicy).parameters
0036:         if "service_owner_sid" in params and "service_owner_sid" not in kwargs:
0037:             kwargs["service_owner_sid"] = "S-1-5-21-1000"
0038:         filtered = {name: value for name, value in kwargs.items() if name in params}
0039:         return self.SecurityPolicy(**filtered)

0041:     def _make_guard(self, **policy_overrides):
0042:         policy = self._make_policy(**policy_overrides)
0043:         return self.SecurityGuard(policy)

0045:     def _call(self, obj, method_names, *args, **kwargs):
0046:         for name in method_names:
0047:             if hasattr(obj, name):
0048:                 method = getattr(obj, name)
0049:                 params = signature(method).parameters
0050:                 if "caller" in params and args:
0051:                     return method(*args, **kwargs)
0052:                 filtered_kwargs = {key: value for key, value in kwargs.items() if key in params}
0053:                 if len(args) == 1 and any(param.kind.name == "KEYWORD_ONLY" for param in params.values()):
0054:                     return method(*args, **filtered_kwargs)
0055:                 return method(*args, **filtered_kwargs)
0056:         self.fail(f"{obj.__class__.__name__} is missing one of the required methods: {method_names}")
