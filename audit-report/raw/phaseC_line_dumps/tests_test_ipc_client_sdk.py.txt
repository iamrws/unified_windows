0018: def _endpoint_to_uri(endpoint: object) -> str:
0019:     if isinstance(endpoint, tuple) and len(endpoint) == 2:
0020:         host, port = endpoint
0021:         return f"tcp://{host}:{port}"
0022:     if isinstance(endpoint, str):
0023:         return endpoint if endpoint.startswith(("tcp://", "pipe://")) else f"tcp://{endpoint}"
0024:     raise AssertionError(f"Unsupported endpoint shape: {endpoint!r}")

0027: def _connect_client(endpoint: str, caller: CallerIdentity) -> AstraWeaveIpcClient:
0028:     client = AstraWeaveIpcClient(
0029:         endpoint=endpoint,
0030:         timeout=5.0,
0031:         default_caller=caller,
0032:         prefer_named_pipe=False,
0033:     )
0034:     last_error: Exception | None = None
0035:     for _ in range(40):
0036:         try:
0037:             client.connect()
0038:             return client
0039:         except Exception as exc:  # pragma: no cover - only used during startup races
0040:             last_error = exc
0041:             time.sleep(0.05)
0042:     if last_error is not None:
0043:         raise last_error
0044:     raise AssertionError("client failed to connect")
