0213: def create_inference_runtime(backend_name: str) -> InferenceRuntime:
0214:     """Create a runtime adapter by stable backend name."""
0215: 
0216:     normalized = (backend_name or "").strip().lower()

0224: def resolve_backend_and_model_name(
0225:     model_name: str,
0226:     runtime_backend: str | None = None,
0227: ) -> tuple[str, str]:
0228:     """Resolve backend + model from explicit selector and/or model-name prefix.
0229: 
0230:     Resolution rules:
0231:     - Explicit runtime backend wins when provided (except `auto`).
0232:     - Otherwise model-name prefix (`ollama:` / `simulation:`) decides.
0233:     - If nothing is specified, default backend is `simulation`.
0234:     """
0235: 
0236:     if not model_name or not model_name.strip():
0237:         raise ApiError(ApiErrorCode.INVALID_ARGUMENT, "model_name must not be empty")
0238: 
0239:     normalized_model = model_name.strip()
0240:     prefixed_backend: str | None = None
0241:     resolved_model = normalized_model
0242:     for prefix, backend in (("ollama:", "ollama"), ("simulation:", "simulation")):
0243:         if normalized_model.lower().startswith(prefix):
0244:             candidate = normalized_model[len(prefix):].strip()
0245:             if not candidate:
0246:                 raise ApiError(ApiErrorCode.INVALID_ARGUMENT, "model_name prefix must include a model identifier")
0247:             prefixed_backend = backend
0248:             resolved_model = candidate
0249:             break
0250: 
0251:     backend_override = None
0252:     if runtime_backend is not None:
0253:         backend_override = runtime_backend.strip().lower()
0254:         if backend_override in {"", "auto"}:
0255:             backend_override = None
0256:         elif backend_override not in {"simulation", "sim", "ollama", "ollama_local"}:
0257:             raise ApiError(ApiErrorCode.INVALID_ARGUMENT, f"unsupported runtime_backend: {runtime_backend}")
0258: 
0259:     if backend_override is not None:
0260:         if backend_override in {"sim", "simulation"}:
0261:             return "simulation", resolved_model
0262:         return "ollama", resolved_model
0263:     if prefixed_backend is not None:
0264:         return prefixed_backend, resolved_model
0265:     return "simulation", resolved_model

0268: def _post_json(url: str, payload: dict[str, Any], timeout_seconds: float) -> dict[str, Any]:
0269:     data = json.dumps(payload).encode("utf-8")
0270:     req = request.Request(
0271:         url,
0272:         data=data,
0273:         headers={"Content-Type": "application/json"},
0274:         method="POST",
0275:     )
0276:     try:
0277:         with request.urlopen(req, timeout=timeout_seconds) as response:
0278:             body = response.read().decode("utf-8")
0279:     except error.HTTPError as exc:  # pragma: no cover - network boundary
0280:         detail = exc.read().decode("utf-8", errors="replace")
0281:         raise ApiError(ApiErrorCode.INTERNAL, f"Ollama request failed with HTTP {exc.code}: {detail}") from exc
0282:     except error.URLError as exc:  # pragma: no cover - network boundary

0284:     except TimeoutError as exc:  # pragma: no cover - network boundary
0285:         raise ApiError(ApiErrorCode.TIMEOUT, "Ollama request timed out") from exc

0287:     try:
