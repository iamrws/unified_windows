0009: class IpcProtocolContractTests(unittest.TestCase):
0010:     @classmethod
0011:     def setUpClass(cls) -> None:
0012:         try:
0013:             cls.protocol = importlib.import_module("astrawave.ipc_protocol")
0014:         except ModuleNotFoundError as exc:  # pragma: no cover - makes the contract explicit
0015:             raise unittest.SkipTest("astrawave.ipc_protocol is not present yet") from exc
0016: 
0017:         try:
0018:             errors = importlib.import_module("astrawave.errors")
0019:             cls.ApiError = errors.ApiError
0020:             cls.ApiErrorCode = errors.ApiErrorCode
0021:         except Exception as exc:  # pragma: no cover - defensive
0022:             raise AssertionError("astrawave.errors must expose ApiError and ApiErrorCode") from exc
0023: 
0024:     def _lookup(self, *names: str):
0025:         for name in names:
0026:             if hasattr(self.protocol, name):
0027:                 return getattr(self.protocol, name)
0028:         self.fail(f"astrawave.ipc_protocol is missing one of: {', '.join(names)}")
0029: 
0030:     def test_request_envelope_validation_accepts_well_formed_input(self) -> None:
0031:         RequestEnvelope = self._lookup("RequestEnvelope")
0032:         CallerEnvelope = self._lookup("CallerEnvelope")
0033:         validate_request_payload = self._lookup("validate_request_payload")
0034: 
0035:         caller = CallerEnvelope(user_sid="S-1-5-21-1000", pid=1001)
0036:         envelope = RequestEnvelope(
0037:             id="req-1",
0038:             method="CreateSession",
0039:             params={},
0040:             caller=caller,
0041:         )
0042:         validated = validate_request_payload(envelope.to_dict())
0043: 
0044:         self.assertEqual(validated.id, "req-1")
0045:         self.assertEqual(validated.method, "CreateSession")
0046:         self.assertEqual(validated.caller.user_sid, caller.user_sid)
0047:         self.assertEqual(validated.caller.pid, caller.pid)
0048: 
0049:     def test_request_envelope_validation_rejects_malformed_input(self) -> None:
0050:         validate_request_payload = self._lookup("validate_request_payload")
0051: 
0052:         malformed_cases = [
0053:             {},
0054:             {"id": "", "method": "CreateSession", "params": {}},
0055:             {"id": "req-1", "method": "", "params": {}},
0056:             {"id": "req-1", "method": "CreateSession"},
0057:             {"id": "req-1", "method": "CreateSession", "params": {"bad": object()}, "caller": None},
0058:             {"id": "req-1", "method": "CreateSession", "params": {}, "caller": {"pid": 1}},
0059:             {"id": "req-1", "method": "CreateSession", "params": {}, "caller": {"user_sid": "", "pid": 1}},
0060:         ]
0061: 
0062:         for payload in malformed_cases:
0063:             with self.subTest(payload=payload):
0064:                 with self.assertRaises(self.ApiError):
0065:                     validate_request_payload(payload)

0067:     def test_error_envelope_maps_api_error_codes_stably(self) -> None:
0068:         ErrorResponse = self._lookup("ErrorResponse")
0069:         ErrorPayload = self._lookup("ErrorPayload")
0070:         error_response = self._lookup("error_response")
0071:         api_error = self.ApiError(self.ApiErrorCode.AUTH_DENIED, "not allowed")
0072: 
0073:         envelope = error_response(api_error, id="req-7")
0074:         payload = envelope.to_dict()
0075: 
0076:         self.assertEqual(payload["id"], "req-7")
0077:         self.assertFalse(payload["ok"])
0078:         self.assertEqual(payload["error"]["code"], self.ApiErrorCode.AUTH_DENIED.value)
0079:         self.assertEqual(payload["error"]["message"], "not allowed")
0080:         self.assertIsInstance(envelope, ErrorResponse)
0081:         self.assertIsInstance(envelope.error, ErrorPayload)
