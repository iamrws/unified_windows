0327: def _distance_to_target_bytes(summary: CandidateSummary, target_ram_bytes: float) -> float:
0328:     peak_private_gb = summary.peak_private_gb
0329:     peak_working_set_gb = summary.peak_working_set_gb
0330:     observed_gb = peak_private_gb if peak_private_gb is not None else peak_working_set_gb

0355: def _summarize_runs(
0356:     candidate: CandidateConfig,
0357:     runs: Sequence[CandidateRun],
0358:     *,
0359:     target_ram_gb: float,
0360: ) -> CandidateSummary:
0361:     run_dicts = [asdict(item) for item in runs]
0362:     success_runs = [item for item in runs if item.ok]
0363: 
0364:     def _mean(values: list[float]) -> float | None:
0365:         if not values:
0366:             return None
0367:         return sum(values) / len(values)
0368: 
0369:     eval_tps_values = [
0370:         float(item.metrics.get("eval_tokens_per_second"))
0371:         for item in success_runs
0372:         if _safe_float(item.metrics.get("eval_tokens_per_second")) is not None
0373:     ]
0374:     end_to_end_tps_values = [
0375:         float(item.metrics.get("end_to_end_tokens_per_second"))
0376:         for item in success_runs
0377:         if _safe_float(item.metrics.get("end_to_end_tokens_per_second")) is not None
0378:     ]
0379:     total_seconds_values = [
0380:         float(item.metrics.get("total_duration_seconds"))
0381:         for item in success_runs
0382:         if _safe_float(item.metrics.get("total_duration_seconds")) is not None
0383:     ]
0384: 
0385:     peak_private_gb = None
0386:     peak_working_set_gb = None
0387:     for item in success_runs:
0388:         private_gb = _safe_float(item.memory_after.get("private_gb"))
0389:         ws_gb = _safe_float(item.memory_after.get("working_set_gb"))
0390:         if private_gb is not None:
0391:             peak_private_gb = private_gb if peak_private_gb is None else max(peak_private_gb, private_gb)
0392:         if ws_gb is not None:
0393:             peak_working_set_gb = ws_gb if peak_working_set_gb is None else max(peak_working_set_gb, ws_gb)
0394:

0396:     ram_target_delta_gb = None
0397:     if observed_gb is not None:
0398:         ram_target_delta_gb = observed_gb - float(target_ram_gb)
0399: 
0400:     run_count = len(runs)
0401:     success_count = len(success_runs)
0402:     success_rate = (float(success_count) / float(run_count)) if run_count > 0 else 0.0
0403: 
0404:     return CandidateSummary(
