0088: def _parse_unittest_log(path: Path) -> tuple[dict[str, str], int | None]:
0089:     statuses: dict[str, str] = {}
0090:     ran_tests: int | None = None
0091:     test_line = re.compile(r"^(test\S+) \(([^)]+)\) \.\.\. (\w+)$")
0092:     ran_line = re.compile(r"^Ran (\d+) tests? in ")
0093: 
0094:     for raw in path.read_text(encoding="utf-8").splitlines():
0095:         line = raw.strip()
0096:         match = test_line.match(line)
0097:         if match:
0098:             test_id = match.group(2)
0099:             status = match.group(3).lower()
0100:             statuses[test_id] = status
0101:             continue
0102:         ran_match = ran_line.match(line)
0103:         if ran_match:
0104:             ran_tests = int(ran_match.group(1))
0105:     return statuses, ran_tests

0108: def _all_ok(statuses: dict[str, str], test_ids: list[str]) -> tuple[bool, list[str]]:
0109:     missing_or_failed = [tid for tid in test_ids if statuses.get(tid) != "ok"]
0110:     return (len(missing_or_failed) == 0), missing_or_failed

0132: def _resolve_path(path_text: str | None, *, root: Path) -> Path | None:
0133:     if path_text is None or not str(path_text).strip():
0134:         return None

0136:     if not path.is_absolute():
0137:         path = (root / path).resolve()

0142:     if kind == "unittest_log":

0146:     if kind == "w13_summary":

0148:     if kind == "operations_report":

0151:         return root / DEFAULT_RELEASE_ARTIFACTS_DIRNAME / f"compliance_manifest_{run_id}.json"

0157: def _find_operation_drill(operations_report: dict[str, Any], drill_id: str) -> dict[str, Any] | None:
0158:     drills = operations_report.get("drills")
0159:     if not isinstance(drills, list):
0160:         return None
0161:     for drill in drills:
0162:         if isinstance(drill, dict) and drill.get("drill_id") == drill_id:
0163:             return drill
0164:     return None
