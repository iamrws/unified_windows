0021:     def setUp(self) -> None:

0038:         self.client = AstraWeaveIpcClient(
0039:             endpoint=endpoint,

0041:             timeout=5.0,
0042:             prefer_named_pipe=False,
0043:         ).connect()
0044:         self.sdk = AstraWeaveSDK(client=self.client)

0046:     def _cleanup(self) -> None:
0047:         for obj in (getattr(self, "sdk", None), getattr(self, "client", None)):
0048:             if obj is None:
0049:                 continue
0050:             close = getattr(obj, "close", None)
0051:             if callable(close):
0052:                 with suppress(Exception):
0053:                     close()
0054:         server = getattr(self, "server", None)
0055:         if server is not None:
0056:             with suppress(Exception):
0057:                 server.stop()

0059:     @staticmethod
0060:     def _endpoint_uri(endpoint: object) -> str:
0061:         if isinstance(endpoint, tuple) and len(endpoint) == 2:
0062:             host, port = endpoint
0063:             return f"tcp://{host}:{port}"
0064:         if isinstance(endpoint, str):
0065:             if endpoint.startswith("\\\\.\\pipe\\"):
0066:                 return f"pipe://{endpoint}"
0067:             return endpoint
0068:         raise AssertionError(f"Unsupported IPC endpoint format: {endpoint!r}")

0091:             step_name="decode",

0093:             max_tokens=8,

0096:         self.assertEqual(run_result["session_id"], session_id)

0100:

0102:         self.assertEqual(residency.session_state.value, "DEGRADED")

0106:         with self.assertRaises(ApiError) as cm:

0109: 
0110:         telemetry_records = self.server.service.telemetry.records

0112:         self.assertGreaterEqual(len(telemetry_records), 4)
0113:         self.assertIn(TelemetryReasonCode.UNKNOWN, reason_codes)
0114:         self.assertIn(TelemetryReasonCode.TRANSFER_PREFETCH, reason_codes)
0115:         self.assertIn(TelemetryReasonCode.FALLBACK_KV_REDUCTION, reason_codes)
