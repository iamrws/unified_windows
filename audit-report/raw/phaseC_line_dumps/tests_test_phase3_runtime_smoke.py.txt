0016: def _endpoint_uri(endpoint: object) -> str:
0017:     if isinstance(endpoint, tuple) and len(endpoint) == 2:
0018:         host, port = endpoint
0019:         return f"tcp://{host}:{port}"
0020:     if isinstance(endpoint, str):
0021:         if endpoint.startswith("\\\\.\\pipe\\"):
0022:             return f"pipe://{endpoint}"
0023:         if endpoint.startswith(("tcp://", "pipe://")):
0024:             return endpoint
0025:         return f"tcp://{endpoint}"
0026:     raise AssertionError(f"Unsupported endpoint shape: {endpoint!r}")

0029: def _wait_for_endpoint(server: AstraWeaveIpcServer) -> str:
0030:     last_endpoint: object | None = None
0031:     for _ in range(40):
0032:         last_endpoint = server.endpoint
0033:         if last_endpoint:
0034:             return _endpoint_uri(last_endpoint)
0035:         time.sleep(0.05)
0036:     raise AssertionError(f"server did not expose an endpoint; last value={last_endpoint!r}")
