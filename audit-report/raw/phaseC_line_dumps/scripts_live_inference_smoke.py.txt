0159: def _validate_option_value(field_name: str, key: str, value: Any) -> Any:
0160:     if key in _POSITIVE_INT_OPTION_KEYS:
0161:         if not isinstance(value, int) or isinstance(value, bool) or value <= 0:
0162:             raise ValueError(f"{field_name}.{key} must be a positive integer")
0163:         return value
0164:     if key in _NONNEGATIVE_INT_OPTION_KEYS:
0165:         if not isinstance(value, int) or isinstance(value, bool) or value < 0:
0166:             raise ValueError(f"{field_name}.{key} must be a non-negative integer")
0167:         return value
0168:     if key in _NONNEGATIVE_FLOAT_OPTION_KEYS:
0169:         if not isinstance(value, (int, float)) or isinstance(value, bool) or float(value) < 0.0:
0170:             raise ValueError(f"{field_name}.{key} must be a non-negative number")
0171:         return float(value)
0172:     if key == "top_p":
0173:         if not isinstance(value, (int, float)) or isinstance(value, bool):
0174:             raise ValueError(f"{field_name}.{key} must be a number")
0175:         numeric_value = float(value)
0176:         if numeric_value <= 0.0 or numeric_value > 1.0:
0177:             raise ValueError(f"{field_name}.{key} must be between 0 and 1")
0178:         return numeric_value
0179:     if key in _OPEN_INTERVAL_FLOAT_OPTION_KEYS:
0180:         if not isinstance(value, (int, float)) or isinstance(value, bool):
