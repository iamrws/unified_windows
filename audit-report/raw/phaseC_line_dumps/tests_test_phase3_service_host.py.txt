0018: def _endpoint_uri(endpoint: object) -> str:
0019:     if isinstance(endpoint, tuple) and len(endpoint) == 2:
0020:         host, port = endpoint
0021:         return f"tcp://{host}:{port}"
0022:     if isinstance(endpoint, str):
0023:         if endpoint.startswith("\\\\.\\pipe\\"):
0024:             return f"pipe://{endpoint}"
0025:         if endpoint.startswith(("tcp://", "pipe://")):
0026:             return endpoint
0027:         return f"tcp://{endpoint}"
0028:     raise AssertionError(f"Unsupported endpoint shape: {endpoint!r}")

0053:     def run_for(self, duration_seconds: float) -> None:
0054:         self.start()
0055:         try:
0056:             deadline = time.monotonic() + duration_seconds
0057:             while time.monotonic() < deadline:
0058:                 time.sleep(0.01)
0059:         finally:
0060:             self.stop()
